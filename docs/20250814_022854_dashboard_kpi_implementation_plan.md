```python
        'procurement_value': df['order_value'].sum(),
        'social_impact_score': calculate_impact_score(df)

# Chart creation functions by type
def create_big_number_card(value, title, delta=None):
    col1, col2 = st.columns([3, 1])
    with col1:
        st.metric(title, f"{value:,}", delta=delta)
    with col2:
        # Add sparkline or mini trend chart
        fig = create_mini_trend_chart(value)
        st.plotly_chart(fig, use_container_width=True)

def create_geographic_map(df, metric_column):
    fig = px.choropleth(
        df, locations='state_code', color=metric_column,
        hover_data=['state_name', 'organization_count'],
        color_continuous_scale='Blues'
    )
    return fig

def create_control_chart(df, metric_column):
    # Statistical process control chart with limits
    mean_val = df[metric_column].mean()
    std_val = df[metric_column].std()
    fig = px.line(df, x='date', y=metric_column)
    fig.add_hline(y=mean_val + 3*std_val, line_dash="dash", line_color="red")
    fig.add_hline(y=mean_val - 3*std_val, line_dash="dash", line_color="red")
    return fig
```
